用户长时间未操作自动退出

Fri, Nov 22, 2019 
app.vue

<template>
  <div id="app" @click="isDo">
    <router-view />
  </div>
</template>

<script>
export default {
  name: "App",
  data() {
    return {
      lastTime: null, //最后一次点击的时间
      currentTime: null, //当前点击的时间
      timeOut: 5 * 60 * 1000, 
      timeId: null
    };
  },
  methods: {
    isDo(event) {
      this.setTimeId();
    },
    setTimeId() {
      clearTimeout(this.timeId);
      this.timeId = setTimeout(() => {
        // 如果有用户登录的
        if(this.$store.state.username) {
          // 退出操作
          this.$store.dispatch("OUT");
          this.$router.push("/login?redirect="+this.$route.fullPath);
        }
      }, this.timeOut);
    }
  },
  created() {
    this.setTimeId();
  }
};
</script>
