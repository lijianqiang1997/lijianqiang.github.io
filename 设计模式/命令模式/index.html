var setCommand = function({
element = null,
eventType = "onclick",
command = null,
commandType = "execute"
}) {
if (element && eventType && typeof element === "object" && eventType in element) {
    if (command && commandType && typeof command === "object" && commandType in command) {
    element[eventType] = command[commandType];
    }
}
};
