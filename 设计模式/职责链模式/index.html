var Chain = function() {
        let _args = arguments;
        return {
          fnArr: [],
          currentIndex: 0,
          addNext: function(fn) {
            return this.fnArr.push(fn), this;
          },
          run: function() {
            let i = this.currentIndex;
            let fnArr = this.fnArr;
            let length = fnArr.length;
            for (; i < length; ++i) {
              this.currentIndex = i;
              let fn = fnArr[i];
              let result = fn(..._args);
              if (result !== true) {
                break;
              }
            }
            if(i === length) {
              this.init();
            }
          },
          init: function() {
            this.currentIndex = 0;
            this.fnArr = [];
          },
          next:function() {
            this.currentIndex ++;
            this.run();
          }
        };
      };
